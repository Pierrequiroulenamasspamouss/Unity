local function _defaultComparator(signalValue, expectedValues, observedValues)
    if signalValue == nil then
        return false
    end

	for i = 0, #observedValues do
        local observedIndex = observedValues[i]
        if signalValue[observedIndex] ~= expectedValues[observedIndex] then
            return false
        end
    end

    return true

end
local function _feedbackLoop(signalName, conditionFunc, expectedValues, observedValues, feedbackFunc)
    local signalValue = qs.waitSignal(signalName, qs.defaultTimeout())

    --print('initial') if type(signalValue) == 'table' then print(signalValue[1]) else print('NOT TABLE') end

    local conditionMet = conditionFunc(signalValue, expectedValues, observedValues)

    if conditionMet == false then
        -- Do arrows and such
        -- qs.debug('_feedbackLoop: DOING ARROWS AND SUCH')
        if feedbackFunc ~= nil then feedbackFunc() end
    end

    while conditionMet == false do
        signalValue = qs.waitSignal(signalName, 0)
        --print('before') if type(signalValue) == 'table' then print(signalValue[1]) else print('NOT TABLE') end
        conditionMet = conditionFunc(signalValue, expectedValues, observedValues)
        --print('after') if type(signalValue) == 'table' then print(signalValue[1]) else print('NOT TABLE') end
    end

    -- Complete
    --qs.debug("_feedbackLoop: COMPLETE")
    --print('result')
    --print(signalValue[1])
    return signalValue
end

--[[
    Returns the key of a value in a table.
--]]
function indexOf(t, value)
    for k, v in pairs(t) do
        if v == value then
            return k
        end
    end
    return nil
end

--[[
    script implementation: waitForDialog {dialogText}
    Waits for the dialog to be closed before continuing execution on the next statement
--]]
function waitForDialog(dialogText)
	qs.showDialog(dialogText)
	qs.waitSignal('PromptReceivedSignal', 0)
end

function waitForLevelUpClose()
	local signalValue = qs.waitSignal('FTUELevelUpOpenSignal', 1) -- wait to see if the level up screen opens
	if signalValue ~= nil then
		-- if the level up screen opened, then wait for it to close again before we proceed
		qs.waitSignal('FTUELevelUpCloseSignal', 0)
	end
end

--[[
    script implementation: waitForBuilding {buildingId} {state}
    Waits for the building with instance id {buildingId} to enter {state}
    before continuing execution on the next statement
    (See qsutil.BuildingState)
--]]
local _waitForBuildingSignal = 'BuildingChangeStateSignal'

function waitForBuilding(buildingId, state, feedbackFunc)
	--print('State: ' .. state)
    if state == 'Construction' then
        qs.openStoreHighlightItem(buildingDefinitionId, false)
    end
    _feedbackLoop(_waitForBuildingSignal, _defaultComparator, {buildingId, state}, {1,2}, feedbackFunc)
end

--[[
    waitForBuildingOfDef
    -----
    Waits for a building of the given definition id to enter the given state.
    -----
    buildingDefId : Id of the building definition to look for
    state         : State for building to enter (See qsutil.BuildingState)
    feedbackFunc  : Custom feedback function.
    -----
    Returns       : The instance of the id whose state has changed.
--]]
function waitForBuildingOfDef(buildingDefId, state, feedbackFunc)
    --qs.debug('Entering waitForBuildingOfDef ' .. buildingDefId .. ', ' .. state)
    local function comparator(signalValue)
        if signalValue == nil then return false end
        local changedInstanceId = signalValue[1]
        local changedDefId = qs.getDefinitionId(changedInstanceId)
        if changedDefId == buildingDefId then
            --qs.debug('def id matches')
            if state == signalValue[2] then
                --qs.debug('waitForBuildingOfDef comparator returning true!')
                return true
            end
        end
        return false
    end

    local signalValue = _feedbackLoop(_waitForBuildingSignal, comparator, nil, nil, feedbackFunc)
    return signalValue[1]
end

--[[
    waitForMinionTasked
    -----
    Yields until a minion has been tasked.
    -----
    buildingId : Id of the building a minion needs to be tasked to.
    feedbackFunc : Custom feedback function.
    -----
    Returns    : Id of the minion tasked.
--]]
function waitForMinionTasked(buildingId, feedbackFunc)
    local function tupleComparator(signalValue, expectedValue, _)
        if signalValue == nil then return false end
        return signalValue[1][1] == expectedValue
    end
    local signalName = 'StartMinionTaskSignal'
    local signalValue = _feedbackLoop(signalName, tupleComparator, buildingId, nil, feedbackFunc)
    --print('right before crash')
    --print(signalValue[1])
    return signalValue[1][2]
end

--[[
    script implementation: waitForMinionSelect {numberMinions}
    Waits for the given number of unique minions to be selected by the player
    before continuing execution on the next statement
--]]
function waitForMinionSelect(numberMinions)
    local signalName = 'SelectMinionSignal'
    local selectedMinions = qs.minionsSelected()
    if (selectedMinions == nil) then
        selectedMinions = {}
    end

    --qs.debug('qsutil.waitForMinionSelect(' .. numberMinions .. ')')
    while #selectedMinions < 3 do
        signalResults = qs.waitSignal(signalName, 0)
        local selectedMinionId = signalResults[1]
        --qs.debug('Minion ' .. selectedMinionId .. ' selected.')
        if not selectedMinions[selectedMinionId] then
            selectedMinions[selectedMinionId] = true
            --qs.debug('Minion added to selected set')
        end
    end

    --qs.debug(str(numberMinions) .. ' minions selected! waitForMinionSelectDone.')
end

--[[
    script implementation: waitForTransaction {toStart|toFinish} transactionId1, transactionId2, ...
    Waits for the given transaction(s) state transition to take place
    before continuing execution on the next statement
--]]
function waitForTransaction(type, ...)
    --qs.debug('waiting for transactions')
    local signalName = 'PostTransactionSignal'
    local idsLeft = {...}
    while #idsLeft > 0 do
        signalResult = qs.waitSignal(signalName, 0)
        if signalResult.Type == type then
            index = qsutil.indexof(idsLeft, signalResult.TransactionId)
            if index ~= nil then
                table.remove(idsLeft, index)
                --qs.debug('Transaction ' .. signalResuilt.TransactionId .. ' received')
            end
        end
    end
    --qs.debug('done waiting for all transactions')
end

--[[
    waitForBuildingMenu
    -----
    Yields until the given building's menu is opened.
    -----
    signalName   : The signal to list for - this depends on the type of building.
                   should be one of ShowBuildingDetailMenuSignal or ShowBuildingDetailMenuSignal
    buildingIds  : The building ids of the buildings that should have their menus opened.
                   Can be a single number or an array.
    feedbackFunc : Custom feedback function.
    -----
    Returns: Id of the building whose menu was opened.
--]]
function waitForBuildingMenu(signalName, buildingIds, feedbackFunc, cleanupFunc)
    if type(buildingIds) ~= 'table' then
            buildingIds = { buildingIds }
    end

    local buildingOpened = nil
    local function comparator(signalValue)
        if signalValue == nil then return false end
        local id = signalValue[1]
        for _, v in pairs(buildingIds) do
            if v == id then
                buildingOpened = v
                return true
            end
        end
        return false
    end

    _feedbackLoop(signalName, comparator, nil, nil, feedbackFunc)
    if cleanupFunc ~= nil then cleanupFunc() end
    return buildingOpened
end

--[[
    Waits for a minion to enter the given state.
    -----
    minionId : Id of the minion to wait for.
               if this parameter is nil, we will wait for any minion.
    state    : The state for the minion to enter ( See MinionState enum)
    -----
    Return Value: The minion which entered the given state.
--]]
function waitForMinion(minionId, state, feedbackFunc, cleanupFunc)
    local signalName = 'MinionStateChangeSignal'

    local expectedValues = nil
    local observedValues = nil
    if minionId == nil then
        expectedValues = {nil, state}
        observedValues = {2}
    else
        expectedValues = {minionId, state}
        observedValues = {1, 2}
    end

    local signalValue = _feedbackLoop(signalName, _defaultComparator, expectedValues, observedValues, feedbackFunc)
    if cleanupFunc ~= nil then cleanupFunc() end
    return signalValue[1]
end

--[[
    Waits for the player to reach a certain level (or returns immediately if already at that level)
    -----
    level 		 : the level the player needs to be.
    feedbackFunc : Custom feedback function.
    -----
    Return Value: the players level after the waited level up.
--]]
function waitForLevel(level, feedbackFunc)
    local signalName = 'LevelUpSignal'
    
    local currentLevel = qs.getPlayerLevel()
    if currentLevel >= level then
    	return currentLevel
    end

    _feedbackLoop(signalName, _defaultComparator, {}, {}, feedbackFunc)
    return qs.getPlayerLevel()
end

function directBuildBuilding(buildingDefinitionId, promptRush, promptTap, forceNewBuilding)

    local storeItemTypeId = qs.getStoreItemIdFromDefinition(buildingDefinitionId)

    instanceId = qs.getInstanceId(buildingDefinitionId)
	if instanceId == 0 or forceNewBuilding then
		instanceId = 0
    	waitForStoreOpened(openStoreHintFunc, openStoreCleanupFunc)
    	qs.highlightStoreItem(buildingDefinitionId, true)
    else
    	qs.panCameraToInstance(instanceId)
    end

    if instanceId == 0 then
    	instanceId = qsutil.waitForBuildingOfDef(buildingDefinitionId, BuildingState.Inactive, coachPlaceBuilding)
    	qs.highlightStoreItem(buildingDefinitionId, false)
    end

    buildingState = qs.getBuildingState(instanceId)
    if buildingState <= BuildingState.Construction then
	    if promptRush then
	    	qs.wait(2)
   		    qs.showDialog(promptRush)
   		end

   		local function revealBuilding()
	       	qs.showMessageBox('ftue_coach_reveal_building')
   		end

		qsutil.waitForBuilding(instanceId, BuildingState.Complete)
	end
	
	if buildingState < BuildingState.Idle then
		if (promptTap) then
    		qsutil.waitForBuilding(instanceId, BuildingState.Idle, revealBuilding)
    	else
    		qsutil.waitForBuilding(instanceId, BuildingState.Idle)
    	end

	    -- wait for crate animation to complete before returning
    	qs.wait(1)
    end

    return instanceId
end
